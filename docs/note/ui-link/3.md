- 1. 删除字符串中所有的 html 标签

```javascript
/**
 * @methods
 * @desc 删除字符串中所有的 html 标签
 * @example '<p>Hello,<strong>world</strong>!</p>' ==> 'Hello,world!'
 * @params {string}
 * @return {String}
 * @author liuxu
 * @date 2023-04-26
 */
function removeHTMLTags(text: string): string {
  if (!text) return "";
  return text.replace(/<[^>]+>/g, "");
}
```

- 2. github gitee 同时支持 ssh

```js

ssh-keygen -t rsa -b 4096 -C "your_email@example.com"
```

```js
// 将 SSH 密钥添加到 SSH 代理：
eval $(ssh-agent -s)
```

```js
ssh-add ~/.ssh/id_rsa_gitee # 替换为您的 Gitee 私钥文件路径
ssh-add ~/.ssh/id_rsa_github # 替换为您的 GitHub 私钥文件路径
```

```js
cat ~/.ssh/id_rsa_gitee.pub # 替换为您的 Gitee 公钥文件路径
cat ~/.ssh/id_rsa_github.pub # 替换为您的 GitHub 公钥文件路径
```

```js
// .ssh/ 新建一个config文件
```

```js
// # Gitee
Host gitee.com
HostName gitee.com
User git
IdentityFile ~/.ssh/id_rsa_gitee
```

```js
// # GitHub
Host github.com
HostName github.com
User git
IdentityFile ~/.ssh/id_rsa_github
```
